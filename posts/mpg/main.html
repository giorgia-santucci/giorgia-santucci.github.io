import pandas as pd
import warnings

warnings.filterwarnings("ignore")
pd.set_option("display.max_columns", 500)
pd.set_option("display.max_rows", 500)
df = pd.read_csv("vehicles.csv")

# Sort the columns
df = df.reindex(sorted(df.columns), axis=1)
# Bring make and model columns to the front
frontcols = ["make", "model", "year"]
df = df[frontcols + [col for col in df.columns if col not in frontcols]]
df = df.apply(lambda x: x.astype("category") if x.dtype == "object" else x)

drive_filter = [
    "4-Wheel or All-Wheel Drive",
    "4-Wheel Drive",
    "All-Wheel Drive",
    "Part-time 4-Wheel Drive",
]
df = df[df["drive"].isin(drive_filter)]


# Remove row Electricity and Hydrogen
fuel_filter = ["Electricity", "Hydrogen"]
df = df[~df["fuelType1"].isin(fuel_filter)]
df.sort_values("feScore", ascending=False, inplace=True)
